> **TASK #179 — Gerçek socket E2E testleri**
>
> Mevcut `e2e/multiplayer.spec.ts` iskeletini gerçek socket.io-client testleriyle doldur. Bu testler Playwright değil, vitest + socket.io-client kullanacak — böylece CI'da çalışır.
>
> Yeni dosya: `artifacts/api-server/src/game/__tests__/multiplayer.test.ts`
>
> ```typescript
> import { createServer } from 'http';
> import { Server } from 'socket.io';
> import { io as Client, Socket } from 'socket.io-client';
> import { describe, it, expect, beforeEach, afterEach } from 'vitest';
> import { setupSocket } from '../socket';
>
> function waitFor(socket: Socket, event: string): Promise<any> {
>   return new Promise(resolve => socket.once(event, resolve));
> }
>
> describe('Multiplayer socket akışı', () => {
>   let httpServer: ReturnType<typeof createServer>;
>   let ioServer: Server;
>   let port: number;
>   let host: Socket;
>   let guest: Socket;
>
>   beforeEach(async () => {
>     httpServer = createServer();
>     ioServer = new Server(httpServer, { cors: { origin: '*' } });
>     setupSocket(ioServer);
>     await new Promise<void>(resolve => httpServer.listen(0, resolve));
>     port = (httpServer.address() as any).port;
>   });
>
>   afterEach(async () => {
>     host?.disconnect();
>     guest?.disconnect();
>     await new Promise<void>(resolve => ioServer.close(() => resolve()));
>   });
>
>   function connect(nickname: string): Socket {
>     return Client(`http://localhost:${port}`, {
>       autoConnect: true,
>       transports: ['websocket'],
>       auth: { nickname }
>     });
>   }
>
>   it('Host oda kurar, guest katılır, ikisi lobide görünür', async () => {
>     host = connect('Host');
>     await waitFor(host, 'connect');
>
>     host.emit('create_room', { nickname: 'Host', settings: {} });
>     const { code } = await waitFor(host, 'state').then((s: any) => s);
>
>     guest = connect('Misafir');
>     await waitFor(guest, 'connect');
>     guest.emit('join_room', { code, nickname: 'Misafir' });
>
>     const state = await waitFor(guest, 'state');
>     const nicknames = state.players.map((p: any) => p.nickname);
>     expect(nicknames).toContain('Host');
>     expect(nicknames).toContain('Misafir');
>   });
>
>   it('Host disconnect → guest geçici host olur', async () => {
>     host = connect('Host');
>     await waitFor(host, 'connect');
>     host.emit('create_room', { nickname: 'Host', settings: {} });
>     const hostState = await waitFor(host, 'state');
>     const { code } = hostState;
>
>     guest = connect('Misafir');
>     await waitFor(guest, 'connect');
>     guest.emit('join_room', { code, nickname: 'Misafir' });
>     await waitFor(guest, 'state');
>
>     // Host bağlantıyı keser
>     host.disconnect();
>
>     // Guest yeni state alır — artık host
>     const newState = await waitFor(guest, 'state');
>     const guestPlayer = newState.players.find((p: any) => p.nickname === 'Misafir');
>     expect(guestPlayer?.isHost).toBe(true);
>   });
>
>   it('Host geri bağlanır — host statüsü iade edilir', async () => {
>     host = connect('Host');
>     await waitFor(host, 'connect');
>     host.emit('create_room', { nickname: 'Host', settings: {} });
>     const { code } = await waitFor(host, 'state');
>
>     guest = connect('Misafir');
>     await waitFor(guest, 'connect');
>     guest.emit('join_room', { code, nickname: 'Misafir' });
>     await waitFor(guest, 'state');
>
>     // Host bağlantıyı keser
>     host.disconnect();
>     await waitFor(guest, 'state'); // temp host atama bekleniyor
>
>     // Host yeniden bağlanır — aynı nickname
>     const reconnectedHost = connect('Host');
>     await waitFor(reconnectedHost, 'connect');
>     reconnectedHost.emit('join_room', { code, nickname: 'Host' });
>
>     const restoredEvent = await waitFor(reconnectedHost, 'host_restored');
>     expect(restoredEvent).toBeDefined();
>
>     const finalState = await waitFor(reconnectedHost, 'state');
>     const hostPlayer = finalState.players.find((p: any) => p.nickname === 'Host');
>     expect(hostPlayer?.isHost).toBe(true);
>
>     reconnectedHost.disconnect();
>   });
>
>   it('4 oyuncuyla oyun başlatılır ve ROLE_SELECT fazına geçilir', async () => {
>     const players: Socket[] = [];
>     let roomCode = '';
>
>     // Host kurar
>     const h = connect('Oyuncu1');
>     await waitFor(h, 'connect');
>     h.emit('create_room', { nickname: 'Oyuncu1', settings: { dayDuration: 60 } });
>     const s = await waitFor(h, 'state');
>     roomCode = s.code;
>     players.push(h);
>
>     // 3 guest katılır
>     for (let i = 2; i <= 4; i++) {
>       const g = connect(`Oyuncu${i}`);
>       await waitFor(g, 'connect');
>       g.emit('join_room', { code: roomCode, nickname: `Oyuncu${i}` });
>       await waitFor(g, 'state');
>       players.push(g);
>     }
>
>     // Host oyunu başlatır
>     h.emit('start_game', { code: roomCode });
>     const gameState = await waitFor(h, 'state');
>     expect(gameState.phase).toBe('ROLE_SELECT');
>     expect(gameState.players).toHaveLength(4);
>
>     players.forEach(p => p.disconnect());
>   });
>
>   it('Oda dolu hatası — 30 oyuncu limitini aşınca error event gelir', async () => {
>     // Bu testi simüle etmek için gameEngine'de max limit 3'e düşür via settings
>     // Gerçek test: 31. oyuncu bağlanmaya çalışır
>     host = connect('Host');
>     await waitFor(host, 'connect');
>     host.emit('create_room', { nickname: 'Host', settings: { maxPlayers: 3 } });
>     const { code } = await waitFor(host, 'state');
>
>     const g1 = connect('G1');
>     g1.emit('join_room', { code, nickname: 'G1' });
>     await waitFor(g1, 'state');
>
>     const g2 = connect('G2');
>     g2.emit('join_room', { code, nickname: 'G2' });
>     await waitFor(g2, 'state');
>
>     // 4. oyuncu (limit 3) hata almalı
>     const g3 = connect('G3');
>     g3.emit('join_room', { code, nickname: 'G3' });
>     const err = await waitFor(g3, 'error');
>     expect(err.code).toBe('ROOM_FULL');
>
>     g1.disconnect(); g2.disconnect(); g3.disconnect();
>   });
> });
> ```
>

>   </Text>
> )}
> ```
>
> Sunucu tarafında: `graveyard_chat` event'i sadece `isAlive === false` oyunculara broadcast edilmeli:
> ```typescript
> socket.on('graveyard_chat', ({ code, message }) => {
>   const room = getRoom(code);
>   const sender = room?.players.find(p => p.socketId === socket.id);
>   if (!sender || sender.isAlive) return; // canlılar yazamaz
>
>   const deadPlayers = room.players.filter(p => !p.isAlive);
>   deadPlayers.forEach(p => {
>     io.to(p.socketId).emit('graveyard_chat', {
>       nickname: sender.nickname,
>       roleId: sender.roleId,
>       text: message,
>       ts: Date.now()
>     });
>   });
> });
> ```
