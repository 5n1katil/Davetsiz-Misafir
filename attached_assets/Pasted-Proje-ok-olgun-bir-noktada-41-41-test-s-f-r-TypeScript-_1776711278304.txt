> Bu en kritik eksik — host "Standart" seçse de tüm 19 rol havuza giriyor.
>
> In `artifacts/api-server/src/game/roles.ts`, update `buildRolePool()`:
>
> ```typescript
> export function buildRolePool(
>   playerCount: number,
>   settings: RoomSettings
> ): string[] {
>   const pkg = settings.rolePackage ?? 'all';
>
>   // Her pakette hangi özel mahalle rolleri aktif
>   const specialMahalleByPackage: Record<string, string[]> = {
>     standard: ['bekci', 'otaci', 'kapici'],
>     advanced: ['bekci', 'otaci', 'kapici', 'falci', 'muhtar', 'muhabir', 'hoca'],
>     all:      ['bekci', 'otaci', 'kapici', 'falci', 'muhtar', 'muhabir', 'hoca', 'tiyatrocu']
>   };
>
>   // Kaos rolleri
>   const kaosByPackage: Record<string, string[]> = {
>     standard:  [],
>     advanced:  ['kirik_kalp', 'dedikoducu'],
>     all:       ['kumarbaz', 'kiskanc_komsu', 'kirik_kalp', 'dedikoducu']
>   };
>
>   // Tarafsız roller
>   const tarafsizByPackage: Record<string, string[]> = {
>     standard:  [],
>     advanced:  [],
>     all:       ['anonim', 'kahraman_dede']
>   };
>
>   const specialMahalle = specialMahalleByPackage[pkg];
>   const kaosRoles      = kaosByPackage[pkg];
>   const tarafsizRoles  = tarafsizByPackage[pkg];
>
>   // Oyuncu sayısına göre çete ve özel slot hesapla
>   const distribution = getRoleDistribution(playerCount);
>   // distribution: { evilCount, tahsildarCount, politikaciCount, ictenCount,
>   //                 kaosCount, tarafsizCount, specialMahalleCount }
>
>   const pool: string[] = [];
>
>   // Çete
>   pool.push('tefeci_basi');
>   for (let i = 0; i < distribution.tahsildarCount; i++) pool.push('tahsildar');
>   if (distribution.politikaciCount) pool.push('sahte_dernek');
>   if (distribution.ictenCount) pool.push('icten_pazarlikli');
>
>   // Özel mahalle (shuffle + slice)
>   const shuffledMahalle = [...specialMahalle].sort(() => Math.random() - 0.5);
>   for (let i = 0; i < Math.min(distribution.specialMahalleCount, shuffledMahalle.length); i++) {
>     pool.push(shuffledMahalle[i]);
>   }
>
>   // Kaos (shuffle + slice)
>   if (distribution.kaosCount > 0 && kaosRoles.length > 0) {
>     const shuffledKaos = [...kaosRoles].sort(() => Math.random() - 0.5);
>     for (let i = 0; i < Math.min(distribution.kaosCount, shuffledKaos.length); i++) {
>       pool.push(shuffledKaos[i]);
>     }
>   }
>
>   // Tarafsız (shuffle + slice)
>   if (distribution.tarafsizCount > 0 && tarafsizRoles.length > 0) {
>     const shuffledTarafsiz = [...tarafsizRoles].sort(() => Math.random() - 0.5);
>     for (let i = 0; i < Math.min(distribution.tarafsizCount, shuffledTarafsiz.length); i++) {
>       pool.push(shuffledTarafsiz[i]);
>     }
>   }
>
>   // Kalan slotları koylu ile doldur
>   while (pool.length < playerCount) pool.push('koylu');
>
>   return pool;
> }
> ```
>
