> **PROBLEM 2 — buildRolePool() dağılım tablosunu kullanmalı**
>
> ```typescript
> export function buildRolePool(playerCount: number, settings: RoomSettings): string[] {
>   const dist = getDistribution(playerCount);
>   const pkg = settings.rolePackage ?? 'all';
>   const disabled = settings.disabledRoles ?? [];
>   const pool: string[] = [];
>
>   // 1. Çete rolleri — zorunlu
>   pool.push('tefeci_basi'); // Her zaman 1 Davetsiz Misafir
>   for (let i = 1; i < dist.davetsizMisafir; i++) pool.push('tefeci_basi');
>   for (let i = 0; i < dist.tahsildar; i++) pool.push('tahsildar');
>   if (dist.politikaci && !disabled.includes('sahte_dernek')) pool.push('sahte_dernek');
>   if (dist.icten && !disabled.includes('icten_pazarlikli')) pool.push('icten_pazarlikli');
>
>   // 2. Özel mahalle rolleri — pakete ve ayara göre
>   const mahallePool = PACKAGE_ROLES[pkg].mahalle.filter(r => !disabled.includes(r));
>   const shuffledMahalle = [...mahallePool].sort(() => Math.random() - 0.5);
>   const mahalleToAdd = shuffledMahalle.slice(0, dist.specialMahalleMax);
>   pool.push(...mahalleToAdd);
>
>   // 3. Kaos rolleri
>   if (dist.kaosMax > 0) {
>     const kaosPool = PACKAGE_ROLES[pkg].kaos.filter(r => !disabled.includes(r));
>     const shuffledKaos = [...kaosPool].sort(() => Math.random() - 0.5);
>     pool.push(...shuffledKaos.slice(0, dist.kaosMax));
>   }
>
>   // 4. Tarafsız roller
>   if (dist.tarafsizMax > 0) {
>     const tarafsizPool = PACKAGE_ROLES[pkg].tarafsiz.filter(r => !disabled.includes(r));
>     const shuffledTarafsiz = [...tarafsizPool].sort(() => Math.random() - 0.5);
>     pool.push(...shuffledTarafsiz.slice(0, dist.tarafsizMax));
>   }
>
>   // 5. Kalan slotları Komşu ile doldur
>   while (pool.length < playerCount) pool.push('komsu');
>
>   // 6. Güvenlik kontrolü — pool tam olmalı
>   if (pool.length !== playerCount) {
>     console.error(`Pool size mismatch: ${pool.length} vs ${playerCount}`);
>     // Düzelt
>     while (pool.length > playerCount) pool.pop();
>     while (pool.length < playerCount) pool.push('komsu');
>   }
>
>   // 7. Denge doğrulama
>   const evilCount = pool.filter(r => ['tefeci_basi','tahsildar','sahte_dernek','icten_pazarlikli'].includes(r)).length;
>   const goodCount = pool.length - evilCount; // kaos ve tarafsız bu hesabın dışında
>   if (evilCount >= goodCount) {
>     console.error(`BALANCE ERROR: evil ${evilCount} >= good ${goodCount} for ${playerCount} players`);
>     // Acil düzeltme: fazla kötü rolleri komşuya çevir
>     let excess = evilCount - goodCount + 1;
>     for (let i = pool.length - 1; i >= 0 && excess > 0; i--) {
>       if (['tahsildar', 'icten_pazarlikli'].includes(pool[i])) {
>         pool[i] = 'komsu';
>         excess--;
>       }
>     }
>   }
>
>   return pool;
> }
> ```

> **PROBLEM 4 — Kazanma koşulu düzeltmesi**
>
> Mevcut kural: `evilAlive >= goodAlive` → çete kazanır.
> Bu yanlış çünkü kaos ve tarafsız oyuncular ne iyiye ne kötüye sayılıyor.
>
> Doğru kural:
>
> ```typescript
> function checkWin(room: Room): string | null {
>   const alive = room.players.filter(p => p.isAlive);
>
>   // Sayılar
>   const evilAlive = alive.filter(p =>
>     ['tefeci_basi','tahsildar','sahte_dernek','icten_pazarlikli'].includes(p.roleId)
>   ).length;
>
>   const goodAlive = alive.filter(p =>
>     ['bekci','otaci','falci','kapici','muhtar','muhabir','tiyatrocu','hoca','komsu'].includes(p.roleId)
>   ).length;
>
>   // Kaos ve tarafsız: ne iyiye ne kötüye sayılmaz
>   // ama kazanma koşullarını etkileyebilir
>
>   // ... (özel kazanma koşulları — Anonim, Kumarbaz vs. değişmedi) ...
>
>   // Çete kazanma — GÜNCELLENDİ:
>   // Kural: evilAlive > 0 VE evilAlive >= goodAlive
>   // Yani: iyi oyuncu kalmadıysa veya eşit/fazlaysa çete kazanır
>   // Kaos ve tarafsız oyuncular bu hesaba GİRMEZ
>   if (evilAlive > 0 && evilAlive >= goodAlive) {
>     return 'kotu';
>   }
>
>   // Mahalle kazanma:
>   // Tüm kötüler yok VE tehlikeli tarafsızlar yok
>   const dangerousNeutrals = alive.filter(p =>
>     ['anonim', 'kahraman_dede'].includes(p.roleId)
>   ).length;
>   if (evilAlive === 0 && dangerousNeutrals === 0) {
>     return 'iyi';
>   }
>
>   return null; // oyun devam eder
> }
> ```
