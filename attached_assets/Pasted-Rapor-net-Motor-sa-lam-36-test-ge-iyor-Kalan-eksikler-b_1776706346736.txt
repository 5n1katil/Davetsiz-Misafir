>     v.language === 'tr-TR' && v.quality === Speech.VoiceQuality.Enhanced

>   // Stop any currently playing speech before starting new
>   const isSpeaking = await Speech.isSpeakingAsync();
>   if (isSpeaking) {
>     await Speech.stop();
