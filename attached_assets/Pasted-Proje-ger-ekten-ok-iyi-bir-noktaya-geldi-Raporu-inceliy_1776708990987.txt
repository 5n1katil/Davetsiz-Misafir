> function buildClientState(room: Room, playerId: string): ClientGameState {
>   const player = room.players.find(p => p.id === playerId);
>   const isHost = player?.isHost ?? false;
>   const isDead = !player?.isAlive;
>
>   return {
>     phase: room.phase,
>     round: room.round,
>     players: room.players.map(p => ({
>       id: p.id,
>       nickname: p.nickname,
>       isAlive: p.isAlive,
>       isHost: p.isHost,
>       isConnected: p.isConnected,
>       // Role only visible to: self, host (for dead players), graveyard members
>       roleId: (
>         p.id === playerId ||
>         (isHost && !p.isAlive) ||

>       // Fake role for Tiyatrocu death — use displayRoleId
>       displayRoleId: p.displayRoleId,
>       voteWeight: p.voteWeight,
>     })),
>     // Host-only fields
>     ...(isHost ? {
>       nightOrderQueue: room.nightOrderQueue,
